<div class="swiper-container image_viewer-slide">
    <div class="swiper-wrapper">
        <div class="swiper-slide obj-fit"><img src="../img/tmp/@img-dummy.png" alt=""></div>
        <div class="swiper-slide obj-fit"><img src="../img/tmp/@img-dummy-x.png" alt=""></div>
        <div class="swiper-slide obj-horizontal-center"><img src="../img/tmp/@img-dummy-landscape.png" alt=""></div>
        <div class="swiper-slide obj-vertical-center"><img src="../img/tmp/@img-dummy-portrait.png" alt=""></div>
    </div>
    <!-- Add Pagination -->
    <div class="swiper-pagination"></div>
    <!-- Add Arrows -->
    <div class="swiper-button swiper-button-next"></div>
    <div class="swiper-button swiper-button-prev"></div>
</div>
        
<script type="text/javascript">
  var isHidden = false;
  var timer;
  var initTimer;
  var imageViewer = new Swiper('.image_viewer-slide', {
      spaceBetween: 0,
      pagination: {
          el: '.swiper-pagination',
          type: 'fraction',
      },
      navigation: {
          nextEl: '.swiper-button-next',
          prevEl: '.swiper-button-prev',
      },
      on: {
          init: function (swiper) {
              initTimer = setTimeout(function () {
                  swiper.pagination.el.classList.add("is-hidden");
                  swiper.navigation.nextEl.classList.add("is-hidden");
                  swiper.navigation.prevEl.classList.add("is-hidden");
                  isHidden = true;
              }, 1500)
          },
          click: function (swiper, event) {
              if (initTimer) {
                  clearTimeout(initTimer);
              }
              if (timer) {
                  clearTimeout(timer);
              }
              timer = setTimeout(function () {
                  if (!isHidden) {
                      swiper.pagination.el.classList.add("is-hidden");
                      swiper.navigation.nextEl.classList.add("is-hidden");
                      swiper.navigation.prevEl.classList.add("is-hidden");
                      isHidden = true;
                  } else {
                      swiper.pagination.el.classList.remove("is-hidden");
                      swiper.navigation.nextEl.classList.remove("is-hidden");
                      swiper.navigation.prevEl.classList.remove("is-hidden");
                      isHidden = false;
                  }
              }, 50)
          },
      }
  });
</script>
