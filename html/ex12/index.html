categorySwiperOpts: {
        slidePerView        : 1,
        autoHeight          : true,
        threshold           : 10,
        pagination          : '.category-slide__pagination',
        paginationType      : 'custom',
        paginationClickable : true,
        onInit: function (swiper) {
          var pagi = $(swiper.paginationContainer[0]).find(".swiper-pagination-bullet");
          pagi.eq(swiper.realIndex).addClass("swiper-pagination-bullet-active");
          pagi.on("click", function () {
            $(this).addClass("swiper-pagination-bullet-active").siblings().removeClass("swiper-pagination-bullet-active");
            swiper.slideTo($(this).index())
          })
        },
      },
