var changeSelectValue = function () {
        var random = Math.floor(Math.random() * 6 + 1);
        var sel = document.querySelector(".select").options;
        var index;
        for (var i in sel) {
            if (sel[i].value === String(random)) {
                index = i;
                break
            }
        }
        document.querySelector(".select").options[index].selected = true;
    }
    document.querySelector(".btn").addEventListener("click", function () {
        changeSelectValue();
    })
